\framebox[\width]{\ tasks.json\ } (Optional) Replace \texttt{.exe} with \texttt{\$\{fileBasenameNoExtension\}}

\begin{lstlisting}
{
    "version": "2.0.0",
    "tasks": [
        {
            "label": "build",
            "type": "shell",
            "command": "cls; g++ '${file}' -std=c++14 -Wall -o .exe"
        },
        {
            "label": "run",
            "type": "shell",
            "command": "cls; Get-Content ./sample.in | ./.exe > ./sample.out"
        },
        {
            "label": "buildrun",
            "type": "shell",
            "command": "cls; g++ '${file}' -std=c++14 -Wall -o .exe; Get-Content ./sample.in | ./.exe > ./sample.out"
        }
    ]
}
\end{lstlisting}
